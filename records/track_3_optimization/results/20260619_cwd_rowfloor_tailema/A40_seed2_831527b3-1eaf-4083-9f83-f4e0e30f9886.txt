def uw_floor_frac(step):
    """Fraction [0,1] of the full u/w-floor active at this step. Default (no env) -> 1.0 always.
    Linear ramp from RAMP_START (frac 0) to RAMP_END (frac 1) avoids the hard-switch jolt."""
    if UW_FLOOR_OFF:
        return 0.0
    if UW_FLOOR_DECAY_START > 0 and step >= UW_FLOOR_DECAY_START:
        # expert(0611): decay the (un-scheduled) floor to 0 across the tail so the LR can anneal into the valley floor.
        return max(0.0, 1.0 - (step - UW_FLOOR_DECAY_START) / max(1, FINAL_TRAIN_STEPS - UW_FLOOR_DECAY_START))
    if UW_FLOOR_RAMP_END > 0:
        if step < UW_FLOOR_RAMP_START:
            return 0.0
        if step >= UW_FLOOR_RAMP_END:
            return 1.0
        return (step - UW_FLOOR_RAMP_START) / max(1, UW_FLOOR_RAMP_END - UW_FLOOR_RAMP_START)
    return 0.0 if step < UW_FLOOR_ON_STEP else 1.0

def _load_data_shard(file: Path):
    header = torch.from_file(str(file), False, 256, dtype=torch.int32) # header is 256 int32
    assert header[0] == 20240520, "magic number mismatch in the data .bin file"
    assert header[1] == 1, "unsupported version"
    num_tokens = int(header[2]) # number of tokens (claimed)
    with file.open("rb", buffering=0) as f:
        tokens = torch.empty(num_tokens, dtype=torch.uint16, pin_memory=True)
        f.seek(256 * 4)
        nbytes = f.readinto(tokens.numpy()) # avoid bytes->array copy
        assert nbytes == 2 * num_tokens, "number of tokens read does not match header"
    return tokens

def distributed_data_generator(filename_pattern: str, batch_size: int, seq_len=1024):
    world_size = dist.get_world_size()
    rank = dist.get_rank()
    files = sorted(Path.cwd().glob(filename_pattern))
    assert batch_size % world_size == 0
    local_batch_size = batch_size // world_size
    file_iter = iter(files)
    tokens, pos = _load_data_shard(next(file_iter)), 0
    while True:
        if pos + batch_size + 1 >= len(tokens):
            tokens, pos = _load_data_shard(next(file_iter)), 0
        buf = tokens[pos + rank * local_batch_size:][:local_batch_size + 1]
        inputs = buf[:-1].to(device="cuda", dtype=torch.int32, non_blocking=True)
        targets = buf[1:].to(device="cuda", dtype=torch.int64, non_blocking=True)
        pos += batch_size
        yield inputs.view(-1, seq_len), targets.view(-1, seq_len)

def norm(x: Tensor):
    return F.rms_norm(x, (x.size(-1),))

class RMSNorm(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.gains = nn.Parameter(torch.ones(dim))

    def forward(self, x):
        return (norm(x.float()) * self.gains).type_as(x)

class Linear(nn.Linear):
    def __init__(self, in_features, out_features):
        super().__init__(in_features, out_features, bias=True)

    def forward(self, x):
        return F.linear(x, self.weight.type_as(x), self.bias.type_as(x))

class Rotary(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        # half-truncate RoPE (w/ base freq tuning)
        angular_freq = (1 / 1024) ** torch.linspace(0, 1, steps=dim//4, dtype=torch.float32)
        self.register_buffer("angular_freq", torch.cat([angular_freq, angular_freq.new_zeros(dim//4)]))

    def forward(self, x_BTHD: Tensor):
        pos = torch.arange(x_BTHD.size(1), dtype=torch.float32, device=x_BTHD.device)
        theta = torch.outer(pos, self.angular_freq)[None, :, None, :]
        cos, sin = theta.cos(), theta.sin()
        x1, x2 = x_BTHD.to(dtype=torch.float32).chunk(2, dim=-1)
        y1 = x1 * cos + x2 * sin
        y2 = x1 * (-sin) + x2 * cos
        return torch.cat((y1, y2), 3).type_as(x_BTHD)

class CausalSelfAttention(nn.Module):
    def __init__(self, dim: int, head_dim=128):
        super().__init__()
        self.num_heads = dim // head_dim
        self.head_dim = head_dim
        hdim = self.num_heads * self.head_dim
        self.q = Linear(dim, hdim)
        self.k = Linear(dim, hdim)
        self.v = Linear(dim, hdim)
        self.proj = Linear(hdim, dim)
        self.rotary = Rotary(head_dim)

    def forward(self, x: Tensor):
        B, T = x.size(0), x.size(1)
        q = self.q(x).view(B, T, self.num_heads, self.head_dim)
        k = self.k(x).view(B, T, self.num_heads, self.head_dim)
        v = self.v(x).view(B, T, self.num_heads, self.head_dim)
        q, k = norm(q), norm(k)
        q, k = self.rotary(q), self.rotary(k)
        y = F.scaled_dot_product_attention(q.transpose(1, 2), k.transpose(1, 2),
                                           v.transpose(1, 2), scale=0.12, is_causal=True).transpose(1, 2)
        y = y.contiguous().view(B, T, self.num_heads * self.head_dim)
        y = self.proj(y)
        return y

class MLP(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        hdim = 4 * dim
        self.fc = Linear(dim, hdim)
        self.proj = Linear(hdim, dim)

    def forward(self, x: Tensor):
        x = self.fc(x)
        x = x.relu().square()
        x = self.proj(x)
        return x

class Block(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.attn = CausalSelfAttention(dim)
        self.mlp = MLP(dim)
        self.norm1 = RMSNorm(dim)
        self.norm2 = RMSNorm(dim)

    def forward(self, x: Tensor):
        x = x + self.attn(self.norm1(x))
        x = x + self.mlp(self.norm2(x))
        return x

class GPT(nn.Module):
    def __init__(self, vocab_size: int, num_layers: int, model_dim: int):
        super().__init__()
        self.embed = nn.Embedding(vocab_size, model_dim).bfloat16()
        self.blocks = nn.ModuleList([Block(model_dim) for _ in range(num_layers)])
        self.proj = Linear(model_dim, vocab_size)
        self.norm1 = RMSNorm(model_dim)
        self.norm2 = RMSNorm(model_dim)

    def forward(self, inputs: Tensor, targets: Tensor):
        x = self.norm1(self.embed(inputs))
        for block in self.blocks:
            x = block(x)
        logits = self.proj(self.norm2(x)).float()
        logits = 15 * logits * (logits.square() + 15**2).rsqrt()
        return F.cross_entropy(logits.view(targets.numel(), -1), targets.view(-1), reduction="sum")

def gram_frobenius_norm_estimate(G: Tensor, keepdim: bool = False, eps: float = 1e-10) -> Tensor:
    X = G.float()
    gram = X.mT @ X if X.size(-2) > X.size(-1) else X @ X.mT
    return gram.norm(dim=(-2, -1), keepdim=keepdim).sqrt().clamp_min(eps)

def _ns_inner(X: Tensor) -> Tensor:
    a, b, c = 2, -1.5, 0.5
    for _ in range(12):
        A = X @ X.mT
        B = b * A + c * A @ A
        X = a * X + B @ X
    return X

def zeropower_via_newtonschulz5(G: Tensor) -> Tensor:
    # Standard Muon Newton-Schulz orthogonalization. (Aurora wide-matrix row-rescale
    # removed -- ablated neutral; this is exactly the AURORA_K=0 path.)
    assert G.ndim >= 2
    X = G.bfloat16()
    if G.size(-2) > G.size(-1):
        X = X.mT
    X = X / gram_frobenius_norm_estimate(X, keepdim=True, eps=1e-7).to(X.dtype)
    X = _ns_inner(X)
    if G.size(-2) > G.size(-1):
        X = X.mT
    return X

def tensor_cosine(a: Tensor, b: Tensor, eps: float = 1e-8) -> Tensor:
    a_f, b_f = a.float(), b.float()
    return (a_f * b_f).sum() / (a_f.norm() * b_f.norm()).clamp_min(eps)

def trust_gate(raw: Tensor, soap: Tensor, grad: Tensor, eps: float = 1e-8) -> Tensor:
    # SOAP is trusted when it still points with raw momentum and is at least as
    # gradient-aligned as raw momentum. This catches stale whitening bases.
    raw_grad = tensor_cosine(raw, grad, eps)
    soap_grad = tensor_cosine(soap, grad, eps)
    soap_raw = tensor_cosine(soap, raw, eps)

    agree_gate = ((soap_raw - ATTN_TRUST_MIN_AGREE) / (1 - ATTN_TRUST_MIN_AGREE)).clamp(0, 1)
    denom = (raw_grad - ATTN_TRUST_MIN_GRAD_ALIGN).clamp_min(eps)
    grad_gate = ((soap_grad - ATTN_TRUST_MIN_GRAD_ALIGN) / denom).clamp(0, 1)
    gate = (agree_gate * grad_gate).clamp(0, 1)
    if ATTN_TRUST_POWER != 1.0:
        gate = gate.pow(ATTN_TRUST_POWER)
    return gate

def early_trust_floor_for_step(step: int) -> float:
    if ATTN_TRUST_FLOOR_FADE_END_STEP <= ATTN_TRUST_FLOOR_END_STEP:
        return 0.0 if step >= ATTN_TRUST_FLOOR_FADE_END_STEP else ATTN_EARLY_TRUST_FLOOR
    if step < ATTN_TRUST_FLOOR_END_STEP:
        return ATTN_EARLY_TRUST_FLOOR
    if step >= ATTN_TRUST_FLOOR_FADE_END_STEP:
        return 0.0
    return ATTN_EARLY_TRUST_FLOOR * (
        ATTN_TRUST_FLOOR_FADE_END_STEP - step
    ) / (ATTN_TRUST_FLOOR_FADE_END_STEP - ATTN_TRUST_FLOOR_END_STEP)

def bounded_trust_gate(gate: Tensor, step: int) -> Tensor:
    floor = early_trust_floor_for_step(step)
    cap = ATTN_EARLY_TRUST_CAP if step < ATTN_TRUST_FLOOR_FADE_END_STEP else 1.0
    return gate.clamp(min=floor, max=cap)

def norm_preserving_blend(raw: Tensor, soap: Tensor, gate: Tensor, eps: float = 1e-8) -> Tensor:
    blended = raw + (soap - raw) * gate.to(raw.dtype)
    raw_norm = gram_frobenius_norm_estimate(raw, eps=eps)
    blended_norm = gram_frobenius_norm_estimate(blended, eps=eps)
    return (blended * (raw_norm / blended_norm).to(blended.dtype)).to(raw.dtype)

def scale_radial_update(update: Tensor, param: Tensor, eps: float = 1e-12) -> Tensor:
    update_f = update.float()
    param_f = param.float()
    denom = (param_f * param_f).sum().clamp_min(eps)
    coeff = (update_f * param_f).sum() / denom
    radial = coeff * param_f
    tangential = update_f - radial
    # p.add_(update, alpha=-lr), so actual movement is -update.
    # Outward movement means (-update) is aligned with p, i.e. coeff < 0.
    radial_scale = torch.where(
        coeff < 0,
        update_f.new_tensor(RADIAL_OUTWARD_SCALE),
        update_f.new_tensor(RADIAL_INWARD_SCALE),
    )
    return (tangential + radial_scale * radial).to(update.dtype)

def target_radius_after_update(param: Tensor, update: Tensor, lr: float, eps: float = 1e-8) -> Tensor:
    param_f = param.float()
    update_f = update.float()
    before_norm = param_f.norm().clamp_min(eps)
    # Use only the radial component's first-order radius change as the intended
    # radius change; the post-step rescale below removes finite tangent drift.
    radial_delta = -lr * (update_f * param_f).sum() / before_norm
    return (before_norm + radial_delta).clamp_min(eps)

def rescale_to_radius(param: Tensor, target_norm: Tensor, eps: float = 1e-8):
    after_norm = param.float().norm().clamp_min(eps)
    param.mul_((target_norm / after_norm).to(param.dtype))

def soap_eigenbasis(mat: Tensor) -> Tensor:
    try:
        _, q = torch.linalg.eigh(mat + 1e-30 * torch.eye(mat.size(0), device=mat.device))
    except RuntimeError:
        _, q = torch.linalg.eigh(mat.double() + 1e-30 * torch.eye(mat.size(0), device=mat.device))
        q = q.float()
    return torch.flip(q, [1])

def soap_basis_qr(row_gg, col_gg, q_row, q_col, exp_avg_sq, lam_row, lam_col):
    row_eig = torch.diag(q_row.T @ row_gg @ q_row)
    row_sort = torch.argsort(row_eig, descending=True)
    q_row = q_row[:, row_sort]
    exp_avg_sq = exp_avg_sq.index_select(0, row_sort)
    lam_row = lam_row.index_select(0, row_sort)
    q_row, _ = torch.linalg.qr(row_gg @ q_row)

    col_eig = torch.diag(q_col.T @ col_gg @ q_col)
    col_sort = torch.argsort(col_eig, descending=True)
    q_col = q_col[:, col_sort]
    exp_avg_sq = exp_avg_sq.index_select(1, col_sort)
    lam_col = lam_col.index_select(0, col_sort)
    q_col, _ = torch.linalg.qr(col_gg @ q_col)
    return q_row, q_col, exp_avg_sq, lam_row, lam_col

def soap_precondition_momentum(update, state, beta2=SOAP_BETA2, eps=1e-8, step=None):
    update_f = update.float()
    if state["q_row"] is None:
        return update
    q_row, q_col = state["q_row"], state["q_col"]
    projected = q_row.T @ update_f @ q_col
    psq = projected.square()
    state["exp_avg_sq"].mul_(beta2).add_(psq, alpha=1 - beta2)
    if SOAP_KL:
        lam_row, lam_col = state["lam_row"], state["lam_col"]
        dr, dc = projected.size(0), projected.size(1)
        # KL-Shampoo eigenvalue-correction: each Kron factor whitened by the other's current inverse
        new_lr = (psq / lam_col.clamp_min(eps).unsqueeze(0)).sum(dim=1) / dc
        new_lc = (psq / lam_row.clamp_min(eps).unsqueeze(1)).sum(dim=0) / dr
        lam_row.mul_(beta2).add_(new_lr, alpha=1 - beta2)
        lam_col.mul_(beta2).add_(new_lc, alpha=1 - beta2)
    s_t = late_damp_frac(step)
    if SOAP_KL and state["soap_step"] > 50:  # identity-warmup: use exp_avg_sq for first 50 steps
        kl = state["lam_row"].unsqueeze(1) * state["lam_col"].unsqueeze(0)
        kl = kl + (s_t * LATE_DAMP * kl.mean() if s_t > 0.0 else 0.0)
        denom = kl.clamp_min(eps * eps).pow(SOAP_DENOM_POWER)
    elif SHAMPOO_MODE:
        # Boza Shampoo: denom = (Gram eigenvalues)^exp instead of SOAP's Adam V. lam = diag(q^T G q) (Rayleigh quotient of
        # the warm-started QR basis -> NOT exact eigvecs, so clamp_min(0) kills roundoff-negatives that would make
        # negative^exp = NaN). eps*eps = absolute NaN-floor for the near-zero early-step spectrum (matches SOAP).
        _lr_eig = (q_row * (state["row_gg"].float() @ q_row)).sum(0).clamp_min(0.0)
        _lc_eig = (q_col * (state["col_gg"].float() @ q_col)).sum(0).clamp_min(0.0)
        _prod = (_lr_eig[:, None] * _lc_eig[None, :]).clamp_min(eps * eps)
        denom = _prod.pow(SHAMPOO_EXP)
        # RELATIVE pseudoinverse fix (Boza's load-bearing detail): floor the denom to SHAMPOO_EPS*max(denom) -> caps 1/denom
        # for the tiny (noise) eigendirections. SHAMPOO_EPS is a fraction of the denom-spectrum max; 1e-12 default -> ~naive.
        denom = denom.clamp_min(SHAMPOO_EPS * denom.max().clamp_min(1e-30))
    else:
        V = state["exp_avg_sq"]
        V = V + (s_t * LATE_DAMP * V.mean() if s_t > 0.0 else 0.0)   # late Tikhonov floor (s_t=0 -> bit-identical)
        denom = V.clamp_min(eps * eps).pow(SOAP_DENOM_POWER)
    precond = q_row @ (projected / denom) @ q_col.T
    precond.mul_(gram_frobenius_norm_estimate(update_f, eps=eps) / gram_frobenius_norm_estimate(precond, eps=eps))
    return precond.to(update.dtype)

def soap_update_preconditioner(grad, state, shampoo_beta=SOAP_BETA2, precondition_frequency=SOAP_PRECONDITION_FREQUENCY):
    grad_f = grad.float()
    state["row_gg"].lerp_(grad_f @ grad_f.T, 1 - shampoo_beta)
    state["col_gg"].lerp_(grad_f.T @ grad_f, 1 - shampoo_beta)
    if state["q_row"] is None:
        state["q_row"] = soap_eigenbasis(state["row_gg"])
        state["q_col"] = soap_eigenbasis(state["col_gg"])
    elif state["soap_step"] > 0 and state["soap_step"] % precondition_frequency == 0:
        state["q_row"], state["q_col"], state["exp_avg_sq"], state["lam_row"], state["lam_col"] = soap_basis_qr(
            state["row_gg"], state["col_gg"], state["q_row"], state["q_col"], state["exp_avg_sq"], state["lam_row"], state["lam_col"]
        )
    state["soap_step"] += 1

def gram_beta_at(step):
    # tail-frozen Gram-beta2 schedule: SOAP_BETA2 until START, then linearly ramp to GRAM_BETA2 by END. -1 = byte-identical.
    if GRAM_BETA2 < 0.0 or step < GRAM_B2_START:
        return SOAP_BETA2
    frac = min(1.0, max(0.0, (step - GRAM_B2_START) / max(1, GRAM_B2_END - GRAM_B2_START)))
    return SOAP_BETA2 + frac * (GRAM_BETA2 - SOAP_BETA2)

def muon_update(update):
    # Newton-Schulz orthogonalization + aspect-ratio scale.
    update = zeropower_via_newtonschulz5(update)
    update *= max(1, update.size(-2) / update.size(-1))**0.5
    return update

# GroupMuon (exp_354, rank-1 design lever): block-diagonal Newton-Schulz on q/k. The whole-matrix polar factor Q=UV^T
# spends unit-singular-value budget on cross-head mixing directions that attention (per-head L2-norm+softmax) never reads.
# Orthogonalizing each contiguous g-head row-block independently realizes more descent mass on the network-visible
# within-head geometry (nuclear-norm subadditivity under row-partition). SHAPE-only (each slice is NS5-Frobenius-normalized
# per-slice) -> survives the radius pin. g = num_heads (=6) is byte-identical to muon_update.
GROUPMUON_HEAD_DIM = 128
GROUPMUON_G = int(os.environ.get("GROUPMUON_G", 0))                        # 0=off; block size in heads, g in {1,2,3,6}
GROUPMUON_G_EARLY = int(os.environ.get("GROUPMUON_G_EARLY", GROUPMUON_G))  # g before T_COARSEN (fine->coarse schedule)
GROUPMUON_T_COARSEN = int(os.environ.get("GROUPMUON_T_COARSEN", 0))        # step to switch g_early->g (0 = constant g)
# Muon+ (exp_356, arXiv 2602.21545): stateless per-COLUMN (and/or row) L2-normalize of the polar factor UV^T, then rms-match.
# Columns are the ONE shape axis the radius-pin + RowFloor + u/w-floor (all row/global) leave untouched -> survives the pin.
# Distinct from in-base NorMuon (row-variance EMA; exp_097 rejected) + pre-polar bilateral (exp_245/246 neutral-worse):
# STATELESS, POST-polar, COLUMN-specific. '' = off = byte-identical.
MUONPLUS = os.environ.get("MUONPLUS", "")                    # '' off; 'col' | 'row' | 'col_row' (both)
MUONPLUS_TARGET = os.environ.get("MUONPLUS_TARGET", "all")   # 'all' (all SOAP 2D) | 'mlp' | 'qk'

def group_muon_update(update, g):
    # block-diagonal NS: orthogonalize each contiguous g-head row-block independently, then restack. Rows are grouped by
    # head (q/k weight is [num_heads*head_dim, d_model], heads contiguous), so reshape [nh*hd, d]->[nh//g, g*hd, d] groups
    # heads [i*g:(i+1)*g] per slice. Requires nh % g == 0. zeropower_via_newtonschulz5 normalizes+orthogonalizes per slice.
    hd = GROUPMUON_HEAD_DIM
    nh = update.size(0) // hd
    blocks = nh // g
    u = update.reshape(blocks, g * hd, update.size(1))
    u = zeropower_via_newtonschulz5(u)
    u = u.reshape(update.size(0), update.size(1))
    u = u * max(1, update.size(-2) / update.size(-1))**0.5
    return u


def psgd_precond(G, state, step):
    # PSGD-Kron whitening: P = (Ql^T Ql) (x) (Qr^T Qr). Preconditioned grad = Ql^T Ql G Qr^T Qr. Every PSGD_UPDATE_FREQ steps,
    # update the factors on the Lie group toward the whitening fixed point AA^T = BB^T (A=Ql G Qr^T, B=Ql^-T V Qr^-1, V~N(0,I)).
    Gf = G.float()
    m, n = Gf.shape
    if "psgd_Ql" not in state:
        state["psgd_Ql"] = torch.eye(m, device=Gf.device, dtype=torch.float32)
        state["psgd_Qr"] = torch.eye(n, device=Gf.device, dtype=torch.float32)
    Ql = state["psgd_Ql"]; Qr = state["psgd_Qr"]
    if step % PSGD_UPDATE_FREQ == 0:
        V = torch.randn_like(Gf)
        A = Ql @ Gf @ Qr.T                                   # [m,n]
        I_m = torch.eye(m, device=Gf.device, dtype=torch.float32)
        I_n = torch.eye(n, device=Gf.device, dtype=torch.float32)
        invQl = torch.linalg.inv(Ql + 1e-8 * I_m)
        invQr = torch.linalg.inv(Qr + 1e-8 * I_n)
        B = invQl.T @ V @ invQr                              # [m,n] = Ql^-T V Qr^-1
        AAl = A @ A.T; BBl = B @ B.T                         # [m,m]
        AAr = A.T @ A; BBr = B.T @ B                         # [n,n]
        gl = (AAl - BBl) / (AAl + BBl).norm().clamp_min(1e-8)
        gr = (AAr - BBr) / (AAr + BBr).norm().clamp_min(1e-8)
        Ql.sub_(PSGD_PRECOND_LR * (gl @ Ql))
        Qr.sub_(PSGD_PRECOND_LR * (gr @ Qr))
    PG = Ql.T @ (Ql @ Gf @ Qr.T) @ Qr                        # P G = Ql^T Ql G Qr^T Qr (the preconditioner shapes the DIRECTION)
    # BUGFIX: normalize PG to muon_update's magnitude (Frobenius = sqrt(min(m,n))*aspect) so lr controls the step like Muon;
    # the raw PG carries the tiny momentum magnitude -> un-normalized steps were magnitude-starved (val 7.2 vs Muon 4.46).
    PG = PG * (PSGD_CLIP * (min(m, n) ** 0.5) * max(1.0, m / n) ** 0.5 / PG.norm().clamp_min(1e-8))
    return PG.to(G.dtype)


class Muon(torch.optim.Optimizer):
    def __init__(self, named_params, lr=0.02, weight_decay=0, mu=0.95):
        assert isinstance(named_params, list) and len(named_params) >= 1
        self.soap_params = {p for n, p in named_params if should_soap_param(n)}
        self.attn_soap_params = {p for n, p in named_params if should_soap_param(n) and is_attn_param(n)}
        self.attn_proj_soap_params = {p for n, p in named_params if should_soap_param(n) and is_attn_proj_param(n)}
        # QK-compositional partner map: q.weight <-> k.weight per block (anisotropic CM whitening)
        _q, _k = {}, {}
        for n, p in named_params:
            if n.endswith(".attn.q.weight"):
                _q[n[:-len(".attn.q.weight")]] = p
            elif n.endswith(".attn.k.weight"):
                _k[n[:-len(".attn.k.weight")]] = p
        self.qk_partner = {}
        for blk, qp in _q.items():
            if blk in _k:
                self.qk_partner[qp] = _k[blk]
                self.qk_partner[_k[blk]] = qp
        # GroupMuon (exp_354): q/k weights eligible for block-diagonal Newton-Schulz
        self.groupmuon_params = {p for n, p in named_params
                                 if n.endswith(".attn.q.weight") or n.endswith(".attn.k.weight")}
        # Muon+ (exp_356): which params get the post-polar column/row L2-normalize
        if MUONPLUS:
            if MUONPLUS_TARGET == "qk":
                self.muonplus_params = {p for n, p in named_params
                                        if n.endswith(".attn.q.weight") or n.endswith(".attn.k.weight")}
            elif MUONPLUS_TARGET == "mlp":
                self.muonplus_params = {p for n, p in named_params if ".mlp." in n}
            else:
                self.muonplus_params = {p for n, p in named_params if should_soap_param(n)}
        else:
            self.muonplus_params = set()
        self.step_count = 0
        params = sorted([p for _, p in named_params], key=lambda x: x.size(), reverse=True)
        defaults = dict(lr=lr, weight_decay=weight_decay, mu=mu)
        super().__init__(params, defaults)

    @torch.no_grad()
    def step(self):
        world_size = dist.get_world_size()
        rank = dist.get_rank()
        for group in self.param_groups:
            params = group["params"]
            params_pad = params + [torch.empty_like(params[-1])] * (world_size - len(params) % world_size)
            for base_i in range(0, len(params), world_size):
                if base_i + rank < len(params):
                    p = params[base_i + rank]
                    state = self.state[p]
                    if len(state) == 0:
                        state["momentum"] = torch.zeros_like(p)
                        if p in self.soap_params:
                            state["exp_avg_sq"] = torch.zeros_like(p, dtype=torch.float32)
                            state["lam_row"] = torch.ones(p.size(0), dtype=torch.float32, device=p.device)
                            state["lam_col"] = torch.ones(p.size(1), dtype=torch.float32, device=p.device)
                            state["row_gg"] = torch.zeros(p.size(0), p.size(0), dtype=torch.float32, device=p.device)
                            state["col_gg"] = torch.zeros(p.size(1), p.size(1), dtype=torch.float32, device=p.device)
                            state["q_row"] = None
                            state["q_col"] = None
                            state["soap_step"] = 0
                    grad = p.grad
                    state["momentum"].lerp_(grad, 1 - group["mu"])
                    momentum_update = grad.lerp(state["momentum"], group["mu"])
                    if CM_ALPHA > 0.0 and p in self.qk_partner:
                        # anisotropic QK-CM: whiten q's momentum by partner k's inverse-sqrt output-Gram (and vice
                        # versa) on the contracted axis. mean-1 -> SHAPE only -> survives radius. Cross-matrix geometry.
                        Wk = self.qk_partner[p].detach().float()                  # partner weight (hdim, d_model)
                        Gk = Wk @ Wk.mT                                           # (hdim, hdim) = W_K W_K^T
                        evals, evecs = torch.linalg.eigh(Gk)
                        Cinv = (evecs * evals.clamp_min(1e-12).pow(-0.5)) @ evecs.mT   # Gram^-1/2
                        Cinv = Cinv / Cinv.diagonal().mean().clamp_min(1e-8)      # mean-1 (shape-only)
                        mw = momentum_update.float()
                        whitened = Cinv @ mw                                      # whiten contracted axis
                        whitened = whitened * (mw.norm() / whitened.norm().clamp_min(1e-8))  # Frobenius-preserve
                        momentum_update = ((1.0 - CM_ALPHA) * mw + CM_ALPHA * whitened).to(momentum_update.dtype)
                    if MUON_SNR > 0.0:
                        # kexue 11593 formula (4): SNR-aware steepest descent. Vg = beta1=beta2=mu EMA of g^2 (gradient
                        # 2nd moment); M~ = M / Vg^(MUON_SNR/2). MUON_SNR=1 => M/sqrt(Vg) = sign(M)/sqrt(1+sigma^2/M^2),
                        # the bounded Adam(b1=b2) update. The element-wise SNR reweight tilts NS5 toward high-SNR
                        # structure (directional -> survives the radius pin). NS5's Frobenius-norm pre-scale absorbs scale.
                        if "grad_v" not in state:
                            state["grad_v"] = grad.float().square()
                        else:
                            state["grad_v"].lerp_(grad.float().square(), 1 - group["mu"])
                        denom = (state["grad_v"] + 1e-12).pow(0.5 * MUON_SNR)
                        momentum_update = (momentum_update.float() / denom).to(momentum_update.dtype)
                    if SNR_MASK and hasattr(p, "_grad_sq"):
                        # cross-replica SNR s=(gbar^2)/v in [0,1] (1=8 shards agree, 0=pure noise); SOAP/NS never see
                        # this 2nd-order cross-shard stat. mean-1 mask re-weights coords WITHIN the matrix -> survives radius.
                        R = dist.get_world_size()
                        v = p._grad_sq.float() / R
                        if "snr_v" not in state:
                            state["snr_v"] = v.clone()
                        else:
                            state["snr_v"].lerp_(v, 0.1)  # beta_v = 0.9
                        s = ((grad.float() / R).square() / (state["snr_v"] + 1e-12)).clamp_(0.0, 1.0)
                        k_eff = SNR_KAPPA * max(0.0, 1.0 - self.step_count / SNR_FADE_END) if SNR_FADE_END > 0 else SNR_KAPPA
                        mhat = s.pow(k_eff)
                        mhat = mhat / mhat.mean().clamp_min(1e-8)
                        momentum_update = (momentum_update.float() * mhat).to(momentum_update.dtype)
                    is_attn_soap = p in self.attn_soap_params
                    use_soap = (p in self.soap_params) and OPT_BASE != "psgd"   # PSGD replaces SOAP for 2D matrices
                    if use_soap:
                        if is_attn_soap:
                            soap_update = soap_precondition_momentum(momentum_update, state, step=self.step_count)
                            if p in self.attn_proj_soap_params and not TRUST_GATE_OFF:
                                gate = bounded_trust_gate(
                                    trust_gate(momentum_update, soap_update, grad),
                                    self.step_count
                                )
                            else:
                                gate = torch.ones((), dtype=torch.float32, device=p.device)
                            momentum_update = norm_preserving_blend(momentum_update, soap_update, gate)
                        else:
                            momentum_update = soap_precondition_momentum(momentum_update, state, step=self.step_count)
                    _muonr = MUONR_ENABLE and self.step_count >= MUONR_START and is_attn_soap
                    if _muonr:
                        # MuonR retraction: rotate W's singular vectors by msign([W^T M]_skew), conserve sigma & |W|_F.
                        Wf = p.float(); Mf = momentum_update.float()
                        Sk = 0.5 * (Wf.t() @ Mf - Mf.t() @ Wf)               # skew-symmetric tangent
                        O = zeropower_via_newtonschulz5(Sk)                   # msign(skew) -> skew & orthogonal (O^2=-I)
                        _muonr_radius = Wf.norm().clamp_min(1e-8)
                        eta = MUONR_SCALE * group["lr"] * (2.0 ** 0.5) * (min(p.shape) ** 0.5) / (2.0 * _muonr_radius)
                        Imat = torch.eye(p.size(1), device=p.device, dtype=torch.float32)
                        p_new = (Wf @ (Imat - eta * O)) / (1.0 + eta * eta) ** 0.5   # orthogonal right-mult: sigma,|W|_F invariant
                        update = ((Wf - p_new) / group["lr"]).to(p.dtype)    # additive form -> p.add_(update,-lr)=p_new; bypass floor/radial
                    elif GROUPMUON_G > 0 and p in self.groupmuon_params:
                        # block-diagonal NS on q/k (fine->coarse schedule). g=6 (num_heads) == muon_update.
                        g_eff = GROUPMUON_G_EARLY if (GROUPMUON_T_COARSEN > 0 and self.step_count < GROUPMUON_T_COARSEN) else GROUPMUON_G
                        update = group_muon_update(momentum_update, g_eff)
                    elif OPT_BASE == "psgd" and p.ndim == 2:
                        update = psgd_precond(momentum_update, state, self.step_count)   # PSGD-Kron preconditioned step (no NS)
                    else:
                        update = muon_update(momentum_update)
                    if MUONPLUS and not _muonr and p in self.muonplus_params:
                        # stateless L2-normalize the polar factor's columns (and/or rows), then rms-match (scale pinned).
                        u = update.float()
                        if "col" in MUONPLUS:
                            u = u / u.norm(dim=0, keepdim=True).clamp_min(1e-8)
                        if "row" in MUONPLUS:
                            u = u / u.norm(dim=1, keepdim=True).clamp_min(1e-8)
                        u = u * (update.float().norm() / u.norm().clamp_min(1e-8))
                        update = u.to(update.dtype)
                    if NS_SPECTRUM_P > 0.0 and use_soap:
                        # spectrum-retention blend (#1): NS5 flattened M_w=U*Sigma_w*V^T to Q=U*V^T (all sing.vals->1),
                        # discarding SOAP's only freq=1 output. Q and M_w share U,V, so the convex blend is a per-
                        # singular-value interpolation toward the SOAP spectrum. rms-match keeps global scale; the
                        # downstream radial+u/w-floor+rescale re-fix only global magnitude so the SHAPE survives.
                        q32 = update.float()
                        mw32 = momentum_update.float()
                        mw32 = mw32 * (q32.norm() / mw32.norm().clamp_min(1e-8))
                        update = ((1.0 - NS_SPECTRUM_P) * q32 + NS_SPECTRUM_P * mw32).to(update.dtype)
                    if WALL_DAMP > 0.0 and use_soap:
                        # AMUSE valley-wall damp: reduce the update's sigma along M_w's TOP singular direction (the
                        # high-curvature/oscillatory wall) from 1 to 1-w_t. Q=U V^T shares (U,V) with M_w; subtract
                        # w_t*u1 v1^T. Warm-started power iter on M_w (mirrors SPEC_WD). rms-match keeps Frobenius
                        # (DIRECTIONAL shape change -> survives the radius pin). Late-gated.
                        w_t = wall_damp_frac(self.step_count)
                        if w_t > 0.0:
                            q32 = update.float()
                            mw32 = momentum_update.float()
                            v1 = state.get("wd_v1")
                            if v1 is None:
                                v1 = torch.ones(mw32.size(1), dtype=torch.float32, device=mw32.device)
                            for _ in range(2):
                                v1 = mw32.T @ (mw32 @ v1)
                                v1 = v1 / v1.norm().clamp_min(1e-12)
                            state["wd_v1"] = v1
                            sd = mw32 @ v1                                    # = sigma1 * u1
                            u1 = sd / sd.norm().clamp_min(1e-12)
                            S = q32 - w_t * torch.outer(u1, v1)              # top-dir sigma: 1 -> 1-w_t
                            S = S * (q32.norm() / S.norm().clamp_min(1e-8))   # rms-match to Q
                            update = S.to(update.dtype)
                    if SV_SHAPE > 0.0 and use_soap:
                        # kexue 11767: reshape Q's flat spectrum to ghat=(sigma_w,i/sigma_w,1)^rho -- keep useful top
                        # (sigma=1), compress the noisy low-curvature TAIL. Q and M_w share (U,V); use eigh(M_w^T M_w)
                        # = V diag(sigma_w^2) V^T to get V and sigma_w. S = U diag(ghat) Vᵀ = Q V diag(ghat) Vᵀ.
                        rho = SV_SHAPE * sv_shape_frac(self.step_count)
                        if rho > 0.0:
                            q32 = update.float()
                            mw32 = momentum_update.float()
                            A = mw32.mT @ mw32
                            lam, V = torch.linalg.eigh(A)
                            s = (lam.clamp_min(1e-12) / lam.max().clamp_min(1e-12)).sqrt()  # sigma_w/sigma_w1 in (0,1]
                            ghat = s.pow(rho)
                            S = ((q32 @ V) * ghat) @ V.mT
                            S = S * (q32.norm() / S.norm().clamp_min(1e-8))   # rms-match to Q
                            update = S.to(update.dtype)
                    if HB_NORM:
                        # SOAP-H/MuonH norm-preserving hyperball (record #19 / PR290): step in the Muon
                        # direction with size lr*||param||, then renormalise ||param|| back to its (init) radius.
                        # Replaces radial damping + u/w-floor + rescale-to-radius; everything else unchanged.
                        pf = p.float(); uf = update.float()
                        p_norm = pf.norm().clamp_min(1e-8)
                        new_p = pf - group["lr"] * uf * p_norm / uf.norm().clamp_min(1e-8)
                        p.copy_((new_p / new_p.norm().clamp_min(1e-8) * p_norm).to(p.dtype))
                        if use_soap:
                            soap_update_preconditioner(grad, state, shampoo_beta=gram_beta_at(self.step_count))
                        continue
                    if CMUON and self.step_count >= CMUON_START:
                        # Cautious-Muon: NS rotation injects coords whose sign opposes the descent grad; zero them and
                        # rescale by inverse unmasked fraction (preserves Frobenius mag -> survives the radius machinery).
                        Mf = (update.float() * grad.float() > 0).float()
                        frac = Mf.mean()
                        masked = (update.float() * Mf * (update.numel() / Mf.sum().clamp_min(1.0))).to(update.dtype)
                        update = torch.where(frac >= 0.5, masked, update)  # >=0.5 unmask floor avoids over-zeroing
                    _manifold = MANIFOLD_MODE and p.ndim == 2 and not _muonr
                    _psgd = OPT_BASE == "psgd" and p.ndim == 2 and not _muonr   # PSGD self-stabilizes -> skip the radius machinery
                    if not _muonr and not _manifold and not _psgd:
                        update = scale_radial_update(update, p)
                    # u/w-floor. SOAP and non-SOAP params can use different floors.
                    p_fro = p.float().norm().clamp_min(1e-8)
                    u_fro = update.float().norm().clamp_min(1e-8)
                    cur_uw = u_fro / p_fro
                    if UW_LOG:
                        full_t = SOAP_TARGET_UW if use_soap else NONSOAP_TARGET_UW
                        self._log_pn = getattr(self, "_log_pn", 0.0) + p_fro.item()
                        self._log_cu = getattr(self, "_log_cu", 0.0) + cur_uw.item()
                        self._log_bd = getattr(self, "_log_bd", 0) + (1 if cur_uw.item() < full_t else 0)
                        self._log_n = getattr(self, "_log_n", 0) + 1
                    if _muonr:
                        target_radius = _muonr_radius   # retraction conserved |W|_F -> rescale_to_radius is a no-op; floor bypassed
                    elif HYPERBALL:
                        if "init_radius" not in state:
                            state["init_radius"] = p.float().norm().clamp_min(1e-8)
                        target_radius = state["init_radius"]
                    else:
                        target_uw = uw_floor_frac(self.step_count) * (SOAP_TARGET_UW if use_soap else NONSOAP_TARGET_UW)
                        if QKDIR and (p in self.qk_partner) and self.step_count >= QKDIR_START and p.ndim == 2:
                            # QK-DIRECTIONAL floor: decompose each head's update into radial (gauge) + tangent (rotation);
                            # floor only the tangent (the network-visible DOF) -> boost rotation, keep gauge as-is.
                            _hd = 128; _nh = p.size(0) // _hd
                            _Pf = p.float().view(_nh, _hd, -1); _Uf = update.float().view(_nh, _hd, -1)
                            _pn2 = _Pf.pow(2).sum(dim=(1, 2), keepdim=True).clamp_min(1e-12)
                            _co = (_Uf * _Pf).sum(dim=(1, 2), keepdim=True) / _pn2          # per-head radial (gauge) coeff
                            _Urad = _co * _Pf; _Utan = _Uf - _Urad
                            _pnorm = _Pf.pow(2).sum(dim=(1, 2), keepdim=True).sqrt()
                            _utn = _Utan.pow(2).sum(dim=(1, 2), keepdim=True).sqrt().clamp_min(1e-12)
                            _fh = torch.clamp(QK_TARGET_MULT * target_uw * _pnorm / _utn, min=1.0).pow(QKDIR_RHO)
                            update = (_Utan * _fh + _Urad).reshape_as(update).to(update.dtype)   # boost rotation, keep gauge
                        elif ROWFLOOR and ROWFLOOR_START <= self.step_count < ROWFLOOR_END and p.ndim == 2:
                            # per-output-row floor: boost each under-updated row to its target update/weight ratio.
                            r_row = p.float().norm(dim=1, keepdim=True).clamp_min(1e-8)
                            s_row = update.float().norm(dim=1, keepdim=True).clamp_min(1e-8)
                            f_row = torch.clamp(target_uw * r_row / s_row, min=1.0).pow(ROWFLOOR_RHO)
                            u2 = update.float() * f_row
                            if ROWFLOOR_FROB:   # FROB=1: redistribute (renorm to the scalar-floor Frobenius budget)
                                g_mag = torch.where(cur_uw < target_uw, target_uw * p_fro, u_fro)
                                u2 = u2 * (g_mag / u2.norm().clamp_min(1e-8))
                            if ROWFLOOR_END_RAMP > 0 and self.step_count > ROWFLOOR_END - ROWFLOOR_END_RAMP:
                                # smooth hand-off: blend per-row update -> scalar-floor update over [END-RAMP, END].
                                beta = (self.step_count - (ROWFLOOR_END - ROWFLOOR_END_RAMP)) / ROWFLOOR_END_RAMP
                                scale = torch.where(cur_uw < target_uw, target_uw * p_fro / u_fro, torch.ones_like(p_fro))
                                u2 = (1.0 - beta) * u2 + beta * (update.float() * scale.float())
                            update = u2.to(update.dtype)   # FROB=0: no renorm -> magnitude grows (the strong version)
                        else:
                            scale = torch.where(cur_uw < target_uw, target_uw * p_fro / u_fro, torch.ones_like(p_fro))
                            update = update * scale.to(update.dtype)
                        if SPEC_WD > 0.0:
                            # #3 spectral weight decay: rank-1 decay of top singular value sigma1(p) via warm-started
                            # power iteration. Fold -SPEC_WD*sigma1*u1*v1^T into update BEFORE target_radius so the
                            # radius bookkeeping redistributes mass to the tail (sigma1 down, Frobenius capacity kept).
                            v1 = state.get("v1")
                            if v1 is None:
                                v1 = torch.ones(p.size(1), dtype=torch.float32, device=p.device)
                            pf = p.float()
                            for _ in range(2):
                                v1 = pf.T @ (pf @ v1)
                                v1 = v1 / v1.norm().clamp_min(1e-12)
                            sd = pf @ v1  # = sigma1 * u1
                            state["v1"] = v1
                            update = update + (SPEC_WD / group["lr"]) * torch.outer(sd, v1).to(update.dtype)
                        target_radius = target_radius_after_update(p, update, group["lr"])
                    if _manifold:
                        # RADICAL PIVOT: drop the radius pin -> step, then reproject each column onto fixed L2 norm s_W.
                        if "manifold_s" not in state:
                            state["manifold_s"] = p.float().norm(dim=0, keepdim=True).clamp_min(1e-8) * MANIFOLD_INIT_MULT
                        p.add_(update, alpha=-group["lr"])
                        _pf = p.float()
                        p.copy_((_pf / _pf.norm(dim=0, keepdim=True).clamp_min(1e-8) * state["manifold_s"]).to(p.dtype))
                    elif _psgd:
                        p.add_(update, alpha=-group["lr"])   # PSGD: plain step, no radius rescale (preconditioner self-stabilizes)
                    else:
                        if CWD > 0.0 and p.ndim == 2:
                            _cwd_mask = (update.float() * p.float() > 0).to(p.dtype)   # coords where -lr*update shrinks |p| (cautious: WD agrees with the update)
                        p.add_(update, alpha=-group["lr"])
                        if CWD > 0.0 and p.ndim == 2 and not CWD_POST:
                            p.mul_(1.0 - (group["lr"] * CWD) * _cwd_mask)             # anisotropic cautious decay BEFORE the radius pin
                        rescale_to_radius(p, target_radius)
                        if CWD > 0.0 and p.ndim == 2 and CWD_POST:
                            p.mul_(1.0 - (group["lr"] * CWD) * _cwd_mask)             # anisotropic cautious decay AFTER the radius pin (preserves magnitude+shape; cc2 winner)
                    if use_soap:
                        soap_update_preconditioner(grad, state, shampoo_beta=gram_beta_at(self.step_count))
                dist.all_gather(params_pad[base_i:base_i + world_size], params_pad[base_i + rank])
        self.step_count += 1
        if UW_LOG and self.step_count % 125 == 0 and getattr(self, "_log_n", 0) > 0:
            print0(f"[uwlog] step={self.step_count} mean_pnorm={self._log_pn/self._log_n:.4f} mean_cur_uw={self._log_cu/self._log_n:.5f} bind_frac={self._log_bd/self._log_n:.3f}")
            self._log_pn = 0.0; self._log_cu = 0.0; self._log_bd = 0; self._log_n = 0


class Adam(torch.optim.Optimizer):
    """Bias-correction-free Adam for the aux (<2D) param groups.

    This is exactly the SOTA's CenterShrinkAdam with center_shrink (rho) = 1.0:
    at rho=1.0 the center-shrink collapses to the identity (mixed == adam_dir and
    the rms-rescale ratio is 1), leaving plain Adam with raw (uncorrected) EMAs.
    NOTE: this intentionally does NOT apply the (1-beta^t) bias correction, so it
    is NOT the same as torch.optim.AdamW. It matches the source under
    AUX_CENTER_SHRINK=1.0 bit-for-bit.
    """
    def __init__(self, params, lr=0.01, betas=(0.8, 0.99), eps=1e-10):
        defaults = dict(lr=lr, betas=betas, eps=eps)
        super().__init__(params, defaults)

    @torch.no_grad()
    def step(self):
        for group in self.param_groups:
            beta1, beta2 = group["betas"]
            eps = group["eps"]
            for p in group["params"]:
                grad = p.grad
                if grad is None:
                    continue
                state = self.state[p]
                if len(state) == 0:
                    state["exp_avg"] = torch.zeros_like(p)
                    state["exp_avg_sq"] = torch.zeros_like(p, dtype=torch.float32)
                exp_avg = state["exp_avg"]
                exp_avg_sq = state["exp_avg_sq"]
                exp_avg.lerp_(grad, 1 - beta1)
                exp_avg_sq.lerp_(grad.float().square(), 1 - beta2)
                adam_dir = exp_avg.float() / (exp_avg_sq.sqrt() + eps)
                p.add_(adam_dir.to(p.dtype), alpha=-group["lr"])


class EMA_Nesterov(torch.optim.Optimizer):
    def __init__(self, params, inner_optimizer, lookahead_stepsize=0, use_scheduled_lookahead_stepsize=True, lookahead_ema=0.9, prefill_steps=0, rest_steps=0):
        if lookahead_stepsize < 0.0:
            raise ValueError("Invalid momentum value: {}".format(lookahead_stepsize))

        super(EMA_Nesterov, self).__init__(params, {})
        self.inner_optimizer = inner_optimizer
        self.use_scheduled_lookahead_stepsize = use_scheduled_lookahead_stepsize
        self.lookahead_stepsize = lookahead_stepsize
        self.lookahead_ema = lookahead_ema
        self.prefill_steps = prefill_steps
        self.rest_steps = rest_steps
        self.it = 0
        self.lookahead_status = False
        self.current_lookahead_stepsize = 0
        self.initialize_buffers()

    def __setstate__(self, state):
        super(EMA_Nesterov, self).__setstate__(state)

    @torch.no_grad()
    def initialize_buffers(self):
        for group in self.param_groups:
            for p in group['params']:
                param_state = self.state[p]
                if 'prev_params' not in param_state:
                    param_state['prev_params'] = (p.clone(), self.it)

                if 'lookahead_buffer' not in param_state:
                    param_state['lookahead_buffer'] = (torch.zeros_like(p), -1)

                if SF_MODE and 'sf_z' not in param_state:
                    # x-primary SF: live param = x; sf_z = fast Muon sequence; sf_x = fp32 x accumulator
                    # (fp32 so the c=1/(t+1) Polyak averaging stays stable late, unlike bf16). z_0 = x_0 = p.
                    param_state['sf_z'] = p.detach().float().clone()
                    param_state['sf_x'] = p.detach().float().clone()

    def get_lr_lambda(self):
        if isinstance(self.inner_optimizer, list):
            lr_lambda = self.inner_optimizer[0].param_groups[0]["lr"] / self.inner_optimizer[0].param_groups[0]["initial_lr"]
        else:
            lr_lambda = self.inner_optimizer.param_groups[0]["lr"] / self.inner_optimizer.param_groups[0]["initial_lr"]
        return lr_lambda

    @torch.no_grad()
    def lookahead_step(self):
        """Performs nesterov's lookahead."""
        if self.use_scheduled_lookahead_stepsize:
            lookahead_stepsize = self.lookahead_stepsize * self.get_lr_lambda()
        else:
            lookahead_stepsize = self.lookahead_stepsize
        self.current_lookahead_stepsize = lookahead_stepsize

        for group in self.param_groups:
            for p in group['params']:

                param_state = self.state[p]

                lookahead = param_state['lookahead_buffer'][0]

                p.add_(lookahead, alpha=lookahead_stepsize)



    @torch.no_grad()
    def accum_lookahead(self):
        """Update nesterov's lookahead direction."""
        lookahead_ema = self.lookahead_ema
        for group in self.param_groups:
            for p in group['params']:
                param_state = self.state[p]
                look = p.add(param_state['prev_params'][0], alpha=-1)

                # update lookahead buffer
                buf = param_state['lookahead_buffer'][0]

                param_state['lookahead_buffer'] = (buf.lerp_(look, 1 - lookahead_ema), self.it) # m^{t+1} = beta * m^t + (1-beta) * look

                # update prev_params buffer
                param_state['prev_params'] = (param_state['prev_params'][0].copy_(p), self.it)


    @torch.no_grad()
    def nesterov_step(self):
        if SF_MODE:
            # write y_t = (1-beta_t) z_t + beta_t x_t into the live params (forward/backward evaluate at y).
            beta = sf_beta_at(self.it)
            for group in self.param_groups:
                for p in group['params']:
                    st = self.state[p]
                    y = st['sf_x'].mul(beta).add_(st['sf_z'], alpha=1.0 - beta)  # fp32 y = beta*x + (1-beta)*z
                    p.copy_(y.to(p.dtype))
            self.lookahead_status = True
            return
        if self.it + 1 > self.prefill_steps and self.it < self.rest_steps and not self.lookahead_status:
            self.lookahead_step()
        else:
            self.current_lookahead_stepsize = 0
        self.lookahead_status = True

    @torch.no_grad()
    def step(self):
        if not self.lookahead_status:
            raise ValueError("optimizer.nesterov_step() should be invoked before model forward pass.")
        if SF_MODE:
            # live params hold y_t (grads taken at y). Move to z_t so the inner step yields z_{t+1}=z_t-gamma*Delta.
            for group in self.param_groups:
                for p in group['params']:
                    p.copy_(self.state[p]['sf_z'].to(p.dtype))
            for opt in self.inner_optimizer:
                opt.step()
            # live params now hold z_{t+1}. Update fp32 z, Polyak-average into x_{t+1}, leave x on the live params.
            c = 1.0 / (self.it + 1)
            if SF_C_FLOOR > 0.0:
                c = max(c, SF_C_FLOOR)
            for group in self.param_groups:
                for p in group['params']:
                    st = self.state[p]
                    st['sf_z'].copy_(p)                              # z_{t+1} (fp32, from live)
                    st['sf_x'].mul_(1.0 - c).add_(st['sf_z'], alpha=c)  # x_{t+1} = (1-c)x_t + c z_{t+1}
                    # eval/next-forward sees z (fast, sharpens with decay, no avg lag) or x (standard SF average);
                    # SF_EVALZ_START: time-varying -- x early (capture averaging mid-gain), z late (decay sharpens).
                    _use_z = SF_EVAL_Z or (SF_EVALZ_START > 0 and self.it >= SF_EVALZ_START)
                    p.copy_((st['sf_z'] if _use_z else st['sf_x']).to(p.dtype))
            self.lookahead_status = False
            self.it += 1
            return
        if isinstance(self.inner_optimizer, list):
            for opt in self.inner_optimizer:
                opt.step()
        else:
            self.inner_optimizer.step()

        self.accum_lookahead()

        self.lookahead_status = False
        self.it += 1

def headmean_antithetic_pair(shape, *, device, layer_idx: int, seed: int, head_dim: int = HEAD_DIM):
    width = shape[-1]
    if width % head_dim != 0:
        s0 = (torch.randint(0, 2, shape, device=device, dtype=torch.float32) * 2 - 1)
        return s0, -s0
    gen = torch.Generator(device=device)
    gen.manual_seed(0xA9170000 + seed * 1009 + layer_idx * 9176)
    heads0 = []
    heads1 = []
    for _ in range(width // head_dim):
        h0 = (torch.randint(0, 2, (head_dim,), device=device, generator=gen, dtype=torch.int64) * 2 - 1).float()
        plus0 = int((h0 > 0).sum().item())
        plus0 = int(round(head_dim / 2 + _CGI_HEAD_MEAN_SHRINK * (plus0 - head_dim / 2)))
        plus0 = max(0, min(head_dim, plus0))
        h0 = torch.cat([
            torch.ones(plus0, device=device, dtype=torch.float32),
            -torch.ones(head_dim - plus0, device=device, dtype=torch.float32),
        ])
        h0 = h0[torch.randperm(head_dim, device=device, generator=gen)]
        plus1 = head_dim - plus0
        h1 = torch.cat([
            torch.ones(plus1, device=device, dtype=torch.float32),
            -torch.ones(head_dim - plus1, device=device, dtype=torch.float32),
        ])
        h1 = h1[torch.randperm(head_dim, device=device, generator=gen)]
        heads0.append(h0)
        heads1.append(h1)
    return torch.cat(heads0).reshape(shape), torch.cat(heads1).reshape(shape)

        if "power_c" in _grp and _grp.get("power_c", 0) > 0:
            _flat = _grp["initial_lr"]; _pc = _grp["power_c"]
            _tcd_gap = (_flat / _pc) ** (1.0 / 1.2)          # t_end - t_cd under the base power 1.2 (preserves cooldown start)
            _grp["power_c"] = _flat / (_tcd_gap ** FINAL_LR_POWER)

# learning rate schedule: stable then decay
def _lr(step, initial_lr, power_c, power=1.0):
    t_end = FINAL_SCHEDULE_STEPS
    flat_lr = initial_lr
    if SF_MODE and not SF_KEEP_DECAY:
        # Schedule-Free: no LR decay. Optional short linear warmup, then CONSTANT lr.
        if SF_WARMUP > 0 and step < SF_WARMUP:
            return flat_lr * (step + 1) / SF_WARMUP
        return flat_lr
    if TAIL_LR_SPLICE and step > TAIL_LR_S0:
        # mid-run preserved bit-for-bit (step<=S0 unchanged); gentler power-1.0 tail, continuous at S0, x ALPHA.
        c_lin = power_c * (t_end - TAIL_LR_S0) ** (power - 1.0)
        downward_lr = TAIL_LR_ALPHA * c_lin * max(0.0, t_end - step)
    else:
        downward_lr = power_c * max(0.0, t_end - step) ** power
    return min(flat_lr, downward_lr)

# v49: v15 Muon mu schedule (warmup 0.85->0.95 over 300 steps, cooldown 0.95->0.85 over last 50)
_MU_MIN = 0.85
_MU_MAX = 0.95
_MU_WARMUP_STEPS = 300
# Muon-momentum cooldown horizon (mu 0.95->0.85 over the last N steps).
_MU_COOLDOWN_STEPS = 200   # tuned (mu-cooldown extended; original 100)
TAIL_MU_END = float(os.environ.get("TAIL_MU_END", _MU_MIN))  # round3 #3: muon mu cooldown target in tail (default _MU_MIN=0.85=SOTA; higher=denoise late)
MU_SCHED_OFF = int(os.environ.get("MU_SCHED_OFF", 0))  # ablation: 1 = constant mu=_MU_MAX (no warmup/cooldown)

def _muon_mu_at_step(step, train_steps):
    if MU_SCHED_OFF:
        return _MU_MAX
    cd_start = train_steps - _MU_COOLDOWN_STEPS
    if step < _MU_WARMUP_STEPS:
        frac = step / max(_MU_WARMUP_STEPS, 1)
        return _MU_MIN + frac * (_MU_MAX - _MU_MIN)
    elif step > cd_start:
        frac = (step - cd_start) / max(_MU_COOLDOWN_STEPS, 1)
        return _MU_MAX - frac * (_MU_MAX - TAIL_MU_END)
    else:
        return _MU_MAX

def set_hparams(step):
    progress = step / FINAL_SCHEDULE_STEPS
    # progress >= 1 is allowed when t_end (FINAL_SCHEDULE_STEPS) < train end: LR is clamped to 0
    # by _lr's max(0, t_end-step) for those tail steps (the harmless frozen coast).
    assert 0 <= progress
    mu = _muon_mu_at_step(step, FINAL_TRAIN_STEPS)
    for opt in optimizers[0].inner_optimizer:
        for group in opt.param_groups:
            group["lr"] = _lr(step, group["initial_lr"], group["power_c"], FINAL_LR_POWER)
    for group in optimizer2.param_groups:
        group["mu"] = mu
    # tail-only eps floor on the radius-free embed/proj Adam denominators (1e-10 default => byte-identical)
    optimizer1.param_groups[0]["eps"] = EMBED_EPS_TAIL if step > EPS_TAIL_START else 1e-10
    optimizer1.param_groups[1]["eps"] = PROJ_EPS_TAIL if step > EPS_TAIL_START else 1e-10
